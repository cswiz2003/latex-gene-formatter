% ======== BADGE MACRO (CIRCULAR AUTO-SIZED) ========
\newcommand{\badge}[1]{%
  \tikz[baseline=(char.base)]{
    \node[
      draw=accent,
      text=accent,
      thick,
      circle,
      inner sep=1.8pt,
      font=\sffamily\bfseries\footnotesize
    ] (char) {#1};
  }%
}
